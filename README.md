## Install

```bash
npm install --save demo-library-hrusikesh
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'demo-library-hrusikesh'
import 'demo-library-hrusikesh/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
